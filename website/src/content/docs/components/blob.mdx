## Imports
```javascript
import { Blob } from "@zephyr3D/react";
```

## Usage

### Standard Blob

<BlobExample />

```javascript
import { Canvas } from "@react-three/fiber";
import React from "react";
import { Blob } from "@zephyr3D/react";
import { OrbitControls } from "@react-three/drei";

export const BlobExample = () => {
  return (
    <div className="flex bg-white/10 h-96 rounded-md shadow-md">
      <Canvas>
        <ambientLight />
        <directionalLight position={[0, 0, 0]} />
        <Blob />
        <OrbitControls />
      </Canvas>
    </div>
  );
};
```

### Different Colours

<BlobColorsExample />

```javascript
import { Canvas } from "@react-three/fiber";
import React from "react";
import { Blob } from "@zephyr3D/react";
import { OrbitControls } from "@react-three/drei";

export const BlobColorsExample = () => {
  return (
    <div className="flex bg-white/10 h-96 rounded-md shadow-md">
      <Canvas camera={{ position: [0, 0, 10] }}>
        <ambientLight />
        <directionalLight position={[0, 0, 0]} />
        <Blob
          color1="skyblue"
          color2="pink"
          meshProps={{ position: [-5, 0, 0] }}
        />
        <Blob
          color1="#636dd8"
          color2="#dbffcc"
          meshProps={{ position: [0, 0, 0] }}
        />
        <Blob
          color1="#ff9393"
          color2="#ffe478"
          meshProps={{ position: [5, 0, 0] }}
        />
        <OrbitControls />
      </Canvas>
    </div>
  );
};
```
